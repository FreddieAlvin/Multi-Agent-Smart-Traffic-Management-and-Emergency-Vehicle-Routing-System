Local incident / roadblock management for the grid environment.

This module defines:

  • Incident: dataclass that represents a temporary event on an edge.
  • EventManager: keeps track of all active incidents and exposes helpers to query blocked edges / nodes and penalties for routing.

 14from typing import Dict, Tuple, List

 17@dataclass
 18class Incident:
 19    """
 20    Represent a single incident (e.g. accident / roadblock) on an edge.
 21
 22    Attributes:
 23        edge:   Normalized edge as a pair of nodes ((x1, y1), (x2, y2)).
 24        severity:
 25            Positive value that encodes how "bad" the incident is.
 26            Large values typically mean the edge should be avoided.
 27        ttl:
 28            Time-to-live in seconds. After this time passes, the incident
 29            is automatically removed.
 30        created_at:
 31            Wall-clock time when the incident was created.
 32        expires_at:
 33            Wall-clock time when the incident should be considered expired.
 34    """
 35    edge: Tuple
 36    severity: float = 1.0
 37    ttl: float = 30.0
 38    created_at: float | None = None
 39    expires_at: float | None = None
 40
 41    def __post_init__(self) -> None:
 42        """Fill in creation / expiration timestamps when not provided."""
 43        now = time.time()
 44        if self.created_at is None:
 45            self.created_at = now
 46        if self.expires_at is None:
 47            self.expires_at = now + self.ttl
 48
 49    def is_active(self) -> bool:
 50        """
 51        Check if this incident is still active.
 52
 53        Returns:
 54            True if the current time is before the expiration time.
 55        """
 56        return time.time() < self.expires_at

 59class EventManager:
 60    """
 61    Manage temporary incidents (roadblocks) on edges of the city graph.
 62
 63    The manager:
 64      * stores incidents keyed by normalized edges;
 65      * automatically removes expired incidents;
 66      * exposes convenience methods for:
 67          - checking if an edge is blocked;
 68          - providing a penalty value for routing;
 69          - enumerating blocked edges and their endpoints.
 70    """
 71
 72    def __init__(self) -> None:
 73        # key: normalized edge ((x1, y1), (x2, y2))
 74        # val: Incident instance
 75        self.incidents: Dict[Tuple, Incident] = {}
 76
 77    # ---------- internal helpers ----------
 78
 79    def _normalize_edge(self, edge: Tuple) -> Tuple:
 80        """
 81        Normalize an edge so that (u, v) and (v, u) map to the same key.
 82
 83        Args:
 84            edge: Tuple of two nodes, each node being a coordinate tuple.
 85
 86        Returns:
 87            A tuple (u, v) with u <= v in lexicographic order.
 88        """
 89        u, v = edge
 90        u, v = tuple(u), tuple(v)
 91        return (u, v) if u <= v else (v, u)
 92
 93    def clear_expired(self) -> None:
 94        """
 95        Remove incidents whose TTL has passed.
 96
 97        This method is called by most public helpers before returning
 98        information, so the map stays clean over time.
 99        """
100        now = time.time()
101        for edge in list(self.incidents.keys()):
102            inc = self.incidents[edge]
103            if inc.expires_at <= now or not inc.is_active():
104                del self.incidents[edge]
105
106    # ---------- public API used elsewhere ----------
107
108    def spawn_incident(self, edge: Tuple, severity: float = 1.0, ttl: float = 30.0) -> None:
109        """
110        Create or refresh an incident on a given edge.
111
112        If an incident on that edge already exists and is active, its
113        severity and expiration time are updated instead of creating
114        a new object.
115
116        Args:
117            edge:     Edge where the incident occurs.
118            severity: Intensity / importance of the incident.
119            ttl:      Time-to-live in seconds.
120        """
121        edge = self._normalize_edge(edge)
122        self.clear_expired()
123
124        if edge in self.incidents and self.incidents[edge].is_active():
125            # refresh existing incident
126            inc = self.incidents[edge]
127            inc.severity = max(inc.severity, severity)
128            inc.expires_at = time.time() + ttl
129        else:
130            self.incidents[edge] = Incident(edge, severity=severity, ttl=ttl)
131
132    def spawn_temporary_block(self, edge: Tuple, ttl: float = 8.0) -> None:
133        """
134        Convenience: spawn a very severe incident on an edge.
135
136        This is used by the random roadblock generator to create edges
137        that should be treated as fully blocked for a short period.
138
139        Args:
140            edge: Edge to temporarily block.
141            ttl:  Time-to-live in seconds for this block.
142        """
143        self.spawn_incident(edge, severity=10.0, ttl=ttl)
144
145    def is_blocked(self, edge: Tuple, threshold: float = 3.0) -> bool:
146        """
147        Check if an edge should be considered blocked.
148
149        Args:
150            edge:      Edge to query.
151            threshold: Minimum severity to count as "blocked".
152
153        Returns:
154            True if there is an incident on the edge with severity
155            greater or equal to the given threshold.
156        """
157        edge = self._normalize_edge(edge)
158        self.clear_expired()
159        inc = self.incidents.get(edge)
160        return bool(inc and inc.severity >= threshold)
161
162    def penalty(self, edge: Tuple) -> float:
163        """
164        Get an extra routing cost for a given edge.
165
166        Args:
167            edge: Edge to query.
168
169        Returns:
170            A positive penalty based on severity if an incident exists,
171            or 0.0 if the edge is clear.
172        """
173        edge = self._normalize_edge(edge)
174        self.clear_expired()
175        inc = self.incidents.get(edge)
176        return float(inc.severity) if inc else 0.0
177
178    def blocked_edges(self, threshold: float = 3.0) -> List[Tuple]:
179        """
180        Return a list of edges that should be drawn as blocked.
181
182        Args:
183            threshold: Minimum severity to be considered blocked.
184
185        Returns:
186            A list of normalized edges with severity above the threshold.
187        """
188        self.clear_expired()
189        return [e for e, inc in self.incidents.items() if inc.severity >= threshold]
190
191    def blocked_nodes(self) -> List[Tuple]:
192        """
193        Return the set of nodes that belong to any blocked edge.
194
195        This is mostly used by the visualizer to plot roadblock icons
196        at the endpoints of blocked edges.
197
198        Returns:
199            List of node coordinate tuples.
200        """
201        nodes: set[Tuple] = set()
202        for u, v in self.blocked_edges():
203            nodes.add(u)
204            nodes.add(v)
205        return list(nodes)
206
207    def __repr__(self) -> str:
208        """Debug representation listing the number of active incidents."""
209        self.clear_expired()
210        return f"EventManager({len(self.incidents)} active: {list(self.incidents.values())})"
