environment.visualization

Visualization module for the Smart Traffic Simulation.

This component is responsible for live rendering of the simulation using Matplotlib. It displays: • road network grid • vehicles • emergency vehicles • traffic lights • hospitals • incidents (blocked edges)

The visualizer runs asynchronously, refreshing at a configurable frequency, pulling data from the shared state dictionary updated by all SPADE agents.

 66class Visualizer:
 67    """
 68    Live visualization engine that renders the simulation environment.
 69
 70    The Visualizer:
 71      • loads sprites from /environment/assets/
 72      • draws the static grid once
 73      • periodically refreshes positions of vehicles, ambulances, lights
 74      • renders incidents as edge markers
 75      • runs as an async loop in parallel with SPADE agents
 76
 77    Attributes:
 78        city (CityEnvironment): The environment containing graph and assets.
 79        shared (dict): Shared dictionary updated by agents with their positions.
 80        refresh (int): Refresh rate (Hz).
 81        period (float): Time per frame (s).
 82        fig, ax: Matplotlib figure and axes used for rendering.
 83    """
 84
 85    def __init__(self, city, shared_state, refresh_hz=5):
 86        """
 87        Initialize the visualization engine.
 88
 89        Args:
 90            city (CityEnvironment): The grid and objects to draw.
 91            shared_state (dict): Real-time positions (vehicles, emergency, lights).
 92            refresh_hz (int): How many times per second to refresh.
 93        """
 94        self.city = city
 95        self.shared = shared_state
 96        self.refresh = max(1, refresh_hz)
 97        self.period = 1.0 / self.refresh
 98        self._last_log = 0.0
 99
100        # Load sprite assets
101        base_path = os.path.join("environment", "assets")
102        self.car_img = mpimg.imread(os.path.join(base_path, "car.png"))
103        self.ev_img = mpimg.imread(os.path.join(base_path, "ambulance.png"))
104        self.hospital_img = mpimg.imread(os.path.join(base_path, "hospital.png"))
105        self.light_img = mpimg.imread(os.path.join(base_path, "traffic_light.png"))
106        self.roadblock_img = mpimg.imread(os.path.join(base_path, "roadblock.png"))
107
108        # Static plot: background + grid
109        self.fig, self.ax = plt.subplots(figsize=(7, 7))
110        self.ax.set_facecolor("#8dc68d")
111
112        pos = {n: n for n in self.city.graph.nodes}
113        nx.draw_networkx_edges(self.city.graph, pos, width=0.2, ax=self.ax)
114
115        # Legend with images
116        handles = ["Traffic Light", "Vehicles", "Emergency", "Hospital", "Incident"]
117        labels = ["Traffic Light", "Vehicles", "Emergency", "Hospital", "Incident"]
118        handler_map = {
119            "Traffic Light": HandlerImage(self.light_img, zoom=0.01),
120            "Vehicles": HandlerImage(self.car_img, zoom=0.03),
121            "Emergency": HandlerImage(self.ev_img, zoom=0.03),
122            "Hospital": HandlerImage(self.hospital_img, zoom=0.01),
123            "Incident": HandlerImage(self.roadblock_img, zoom=0.03),
124        }
125
126        legend = self.ax.legend(
127            handles=handles,
128            labels=labels,
129            handler_map=handler_map,
130            loc="upper left",
131            bbox_to_anchor=(1.02, 1.0),
132            borderaxespad=0.0,
133            frameon=True,
134        )
135        legend.get_frame().set_facecolor("#d3d3d3")
136        legend.get_frame().set_edgecolor("none")
137
138        self.ax.set_title("Smart Traffic — Live")
139        self.ax.set_xlim(-1, self.city.width)
140        self.ax.set_ylim(-1, self.city.height)
141        self.ax.set_aspect("equal", adjustable="box")
142        plt.tight_layout()
143
144        # Runtime artists
145        self._img_artists = []
146        self._roadblock_artists = []
147
148    async def update_loop(self):
149        """
150        Asynchronous refresh loop that updates the visual state continuously.
151
152        This method:
153            • calls `_update()` once per refresh period
154            • redraws only dynamic elements
155            • keeps the Matplotlib window alive without blocking SPADE
156        """
157        plt.show(block=False)
158        while True:
159            await asyncio.sleep(self.period)
160            self._update()
161            self.fig.canvas.draw_idle()
162            self.fig.canvas.flush_events()
163
164    def _update(self):
165        """
166        Redraw all dynamic elements (vehicles, ambulances, incidents, lights).
167
168        This method fetches the latest shared state values updated by all agents,
169        removes previous frame artifacts, and redraws icons at updated positions.
170        """
171        # Read state -----------------------
172        vpos = list(self.shared.get("vehicles", {}).values())
173        v_arr = np.array(vpos, float).reshape(-1, 2) if vpos else np.empty((0, 2))
174
175        epos = list(self.shared.get("emergency", {}).values())
176        e_arr = np.array(epos, float).reshape(-1, 2) if epos else np.empty((0, 2))
177
178        hpos = list(self.city.hospitals.values())
179        lpos = list(self.shared.get("lights", []))
180
181        rnodes = self.city.event_manager.blocked_nodes()
182        redges = self.city.event_manager.blocked_edges()
183
184        # Clear previous frame ------------
185        for art in self._img_artists:
186            art.remove()
187        for art in self._roadblock_artists:
188            art.remove()
189        self._img_artists.clear()
190        self._roadblock_artists.clear()
191
192        # Hospitals
193        for x, y in hpos:
194            icon = OffsetImage(self.hospital_img, zoom=0.02)
195            ab = AnnotationBbox(icon, (x, y), frameon=False, zorder=10)
196            self.ax.add_artist(ab)
197            self._img_artists.append(ab)
198
199        # Traffic lights
200        for x, y in lpos:
201            icon = OffsetImage(self.light_img, zoom=0.01)
202            ab = AnnotationBbox(icon, (x, y), frameon=False, zorder=10)
203            self.ax.add_artist(ab)
204            self._img_artists.append(ab)
205
206        # Vehicles
207        for x, y in v_arr:
208            icon = OffsetImage(self.car_img, zoom=0.03)
209            ab = AnnotationBbox(icon, (x, y), frameon=False, zorder=12)
210            self.ax.add_artist(ab)
211            self._img_artists.append(ab)
212
213        # Emergency vehicles
214        for x, y in e_arr:
215            icon = OffsetImage(self.ev_img, zoom=0.03)
216            ab = AnnotationBbox(icon, (x, y), frameon=False, zorder=13)
217            self.ax.add_artist(ab)
218            self._img_artists.append(ab)
219
220        # Incidents (edges + icon)
221        for (u, v) in redges:
222            # red line marking blocked edge
223            art = self.ax.plot(
224                [u[0], v[0]],
225                [u[1], v[1]],
226                color="red",
227                linewidth=2,
228                zorder=11,
229                alpha=0.6,
230            )[0]
231            self._roadblock_artists.append(art)
232
233            # icon at midpoint
234            mx = (u[0] + v[0]) / 2
235            my = (u[1] + v[1]) / 2
236            icon = OffsetImage(self.roadblock_img, zoom=0.025)
237            ab = AnnotationBbox(icon, (mx, my), frameon=False, zorder=12)
238            self.ax.add_artist(ab)
239            self._img_artists.append(ab)
240
241        # Log heartbeat
242        now = time.time()
243        if now - self._last_log > 0.5:
244            print(
245                f"[VIS] vehicles={len(v_arr)} "
246                f"emergency={len(e_arr)} "
247                f"hospitals={len(hpos)} "
248                f"traffic_lights={len(lpos)} "
249                f"blocked_edges={len(redges)}"
250            )
251            self._last_log = now
