traffic_lights.py

Implements the TrafficLightAgent, a SPADE agent that simulates an adaptive traffic light in a grid-based smart-city environment.

Each traffic light:

  • Alternates between two phases: Phase 0: North/South movement allowed Phase 1: East/West movement allowed
  • Adapts its phase duration based on local road congestion
  • Responds to two types of requests:
    • priority_request (emergency vehicles)
    • passage_request (regular vehicles)
  • Integrates with the EventManager to block edges affected by incidents
  • Follows a Contract Netâ€“style reply semantics (accept/reject proposals)

This file defines: direction() â†’ Infer direction of movement from grid coordinates. TrafficLightAgent â†’ SPADE agent controlling a single traffic light.

 37def direction(from_pos, to_pos):
 38    """
 39    Infer movement direction between two grid coordinates.
 40
 41    Args:
 42        from_pos (tuple[int, int]): Current position (x, y).
 43        to_pos (tuple[int, int]): Next position (x, y).
 44
 45    Returns:
 46        str | None:
 47            "N", "S", "E", or "W" depending on movement.
 48            None if movement is diagonal or zero.
 49    """
 50    (x1, y1) = from_pos
 51    (x2, y2) = to_pos
 52    dx = x2 - x1
 53    dy = y2 - y1
 54
 55    if dx == 1 and dy == 0:
 56        return "E"
 57    if dx == -1 and dy == 0:
 58        return "W"
 59    if dx == 0 and dy == 1:
 60        return "N"
 61    if dx == 0 and dy == -1:
 62        return "S"
 63
 64    return None

 68class TrafficLightAgent(agent.Agent):
 69    """
 70    Traffic light controller agent.
 71
 72    This agent autonomously alternates between NS and EW phases, adjusts the
 73    duration of each phase according to observed traffic density, and processes
 74    vehicle passage requests.
 75
 76    Attributes:
 77        city (CityEnvironment): The global environment object.
 78        shared (dict): Shared state dictionary for visualization.
 79        position (tuple[int, int] | None): Grid coordinate of the light.
 80        phase (int): Current phase (0=NS, 1=EW).
 81        phase_duration (float): Adaptive duration of the current phase.
 82        _last_switch (float): Time of last phase toggle.
 83    """
 84
 85    def __init__(self, jid, password, city_env=None, shared=None):
 86        super().__init__(jid, password)
 87        self.city = city_env
 88        self.shared = shared or {}
 89        self.position = None
 90        self.phase_duration = PHASE_DURATION
 91        self.phase = 0
 92        self._last_switch = 0.0

 94    class LightBehaviour(behaviour.CyclicBehaviour):
 95        """
 96        Cyclic behaviour controlling the traffic-light loop.
 97
 98        Responsibilities:
 99            1. Adapt phase duration using local traffic density.
100            2. Toggle phase when time expires.
101            3. Process passage requests (regular or emergency).
102            4. Use Contract Net semantics for responses.
103        """
104
105        async def on_start(self):
106            """Initializes light phase and timestamp."""
107            self.agent.phase = 0
108            self.agent._last_switch = asyncio.get_event_loop().time()
109            print(
110                f"[Traffic Light {self.agent.jid}] ðŸš¦ started in phase "
111                f"{self.agent.phase}"
112            )
113
114        async def run(self):
115            """Main control loop executed repeatedly."""
116            now = asyncio.get_event_loop().time()
117
118            # -------------------------------------------------------
119            # 1) Adapt duration using local traffic density
120            # -------------------------------------------------------
121            if getattr(self.agent, "city", None) and self.agent.position:
122                try:
123                    rho = self.agent.city.occupancy.local_density(
124                        self.agent.position,
125                        radius=2,
126                    )
127                    self.agent.phase_duration = max(
128                        3.0, min(8.0, 3.0 + 5.0 * float(rho))
129                    )
130                except Exception as e:
131                    print(f"[Traffic Light {self.agent.jid}] âš ï¸ density adapt error: {e}")
132                    self.agent.phase_duration = PHASE_DURATION
133
134            # -------------------------------------------------------
135            # 2) Toggle phase if duration expired
136            # -------------------------------------------------------
137            if now - self.agent._last_switch >= self.agent.phase_duration:
138                self.agent.phase = 1 - self.agent.phase
139                self.agent._last_switch = now
140                print(
141                    f"[Traffic Light {self.agent.jid}] ðŸ”„ Switched to phase {self.agent.phase} "
142                    f"(allowed={ALLOWED_BY_PHASE[self.agent.phase]}, "
143                    f"duration={self.agent.phase_duration:.1f}s)"
144                )
145
146            # -------------------------------------------------------
147            # 3) Handle incoming passage requests
148            # -------------------------------------------------------
149            msg = await self.receive(timeout=0.1)
150            if not msg:
151                return
152
153            mtype = msg.metadata.get("type") if msg.metadata else None
154
155            # Parse body safely
156            try:
157                payload = json.loads(msg.body)
158                from_pos = tuple(payload["from"])
159                to_pos = tuple(payload["to"])
160            except Exception:
161                print(
162                    f"[Traffic Light {self.agent.jid}] âš ï¸ Invalid message body: "
163                    f"{msg.body}"
164                )
165                await self._reply(msg, granted=False, reason="invalid_body")
166                return
167
168            # Emergency: always granted
169            if mtype == "priority_request":
170                print(
171                    f"[Traffic Light {self.agent.jid}] ðŸš‘ Priority request "
172                    f"{from_pos}->{to_pos} -> GRANTED"
173                )
174                await self._reply(msg, granted=True, reason="emergency")
175                return
176
177            # Non-emergency vehicle
178            d = direction(from_pos, to_pos)
179            if d is None:
180                print(
181                    f"[Traffic Light {self.agent.jid}] âŒ Unknown direction "
182                    f"{from_pos}->{to_pos}"
183                )
184                await self._reply(msg, granted=False, reason="unknown_direction")
185                return
186
187            allowed_dirs = ALLOWED_BY_PHASE[self.agent.phase]
188            granted = d in allowed_dirs
189            reason = "phase_check"
190
191            # Incident check (blocked edge)
192            if getattr(self.agent.city, "event_manager", None):
193                edge = (from_pos, to_pos)
194                if self.agent.city.event_manager.is_blocked(edge):
195                    granted = False
196                    reason = "blocked_by_incident"
197                    print(
198                        f"[Traffic Light {self.agent.jid}] ðŸš§ Edge {edge} blocked by incident -> DENIED"
199                    )
200
201            print(
202                f"[Traffic Light {self.agent.jid}] ðŸš— Request {from_pos}->{to_pos} "
203                f"dir={d} phase={self.agent.phase} allowed={allowed_dirs} -> "
204                f"{'GRANTED' if granted else 'DENIED'} (reason={reason})"
205            )
206
207            await self._reply(msg, granted=granted, reason=reason)

209        async def _reply(self, msg, granted: bool, reason: str = ""):
210            """
211            Send a Contract Netâ€“style reply to a vehicle or ambulance.
212
213            Args:
214                msg (Message): Incoming request message.
215                granted (bool): Whether passage is allowed.
216                reason (str): Explanation for logging.
217            """
218            reply = Message(to=str(msg.sender))
219            reply.set_metadata("type", "passage_reply")
220            reply.set_metadata("protocol", "contract-net")
221
222            mtype = msg.metadata.get("type") if msg.metadata else None
223            if mtype == "priority_request":
224                reply.set_metadata("performative", "accept-proposal")
225            else:
226                reply.set_metadata(
227                    "performative",
228                    "accept-proposal" if granted else "reject-proposal",
229                )
230
231            reply.set_metadata("granted", "true" if granted else "false")
232            if reason:
233                reply.set_metadata("reason", reason)
234
235            reply.body = "granted" if granted else "denied"
236            await self.send(reply)
237            await asyncio.sleep(1.5)

245        """
246        print(f"Traffic Light Agent {self.jid} ready")
247
248        if self.city:
249            lid = str(self.jid).split("@")[0]
250            self.position = self.city.traffic_lights.get(lid)
251
252        self.phase_duration = PHASE_DURATION
253        self.add_behaviour(self.LightBehaviour())
