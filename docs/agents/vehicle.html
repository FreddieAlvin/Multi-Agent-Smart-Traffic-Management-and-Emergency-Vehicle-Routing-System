 39def manhattan(a, b):
 40    """
 41    Compute Manhattan distance between two grid points.
 42
 43    Args:
 44        a (tuple[int, int]): First point (x, y).
 45        b (tuple[int, int]): Second point (x, y).
 46
 47    Returns:
 48        int: |x1 - x2| + |y1 - y2|.
 49    """
 50    return abs(a[0] - b[0]) + abs(a[1] - b[1])

 55    Normal (non-emergency) vehicle agent.

 57    Responsibilities:
 58      - Choose random, reasonably distant destinations.

 61            * congestion (occupancy) density,
 62            * penalties due to incidents/roadblocks.

103    def _dynamic_weight(self, u, v, d):
104        """
105        Compute the dynamic weight of an edge for A*.
106
107        Combines:
108          - base weight of the edge,
109          - congestion density,
110          - incident/roadblock penalties on edge and target node.
111
112        Args:
113            u (tuple[int, int]): Source node.
114            v (tuple[int, int]): Target node.
115            d (dict): Edge attributes.
116
117        Returns:
118            float: Effective cost for the A* search.
119        """
120        base = d.get("weight", 1.0)
121
122        # Occupancy (congestion)
123        try:
124            occ = self.city.occupancy.edge_density(u, v)
125        except Exception:
126            occ = 0.0
127
128        # Incident penalties
129        pen = 0.0
130        try:
131            pen = self.city.event_manager.edge_penalty(u, v)
132        except Exception:
133            pass
134
135        # Avoid entering blocked nodes
136        try:
137            if self.city.event_manager.is_node_blocked(v):
138                pen += 9999.0
139        except Exception:
140            pass
141
142        return base + 0.6 * occ + pen

160    def _plan_to(self, dest):
161        """
162        Plan an A* route to a given destination.
163
164        Records replans in metrics and updates:
165          - self.path
166          - self.goal
167          - self.steps_since_plan
168
169        Args:
170            dest (tuple[int, int]): Target node.
171        """
172        metrics = getattr(self.city, "metrics", None)
173        if metrics is not None:
174            metrics.log_replan(self.label)
175
176        try:
177            self.path = nx.astar_path(
178                self.city.graph,
179                self.position,
180                dest,
181                heuristic=manhattan,
182                weight=lambda u, v, d: self._dynamic_weight(u, v, d)
183            )
184            self.goal = dest
185            self.steps_since_plan = 0
186            print(f"[{self.label}]  planned path to {dest} (len={len(self.path)})")
187        except Exception as e:
188            print(f"[{self.label}]  plan failed ({e}); fallback to random move")
189            self.path = []
190            self.goal = None

192    def _step_along_path(self):
193        """
194        Execute one step along the planned path.
195
196        Enforces:
197          - path consistency,
198          - edge/node block checks,
199          - fallback to idle if movement impossible.
200
201        Returns:
202            tuple[tuple[int, int], tuple[int, int]]:
203                (old_position, new_position)
204        """
205        if not self.path or self.path[0] != self.position:
206            if self.path and self.position in self.path:
207                idx = self.path.index(self.position)
208                self.path = self.path[idx:]
209            else:
210                return self._move_randomly()
211
212        if len(self.path) >= 2:
213            curr = self.path[0]
214            nxt = self.path[1]
215
216            # Blocked edge?
217            try:
218                if self.city.event_manager.is_blocked((curr, nxt)):
219                    print(f"[{self.label}]  Edge {(curr, nxt)} blocked  replanning")
220                    self.path = []
221                    self.goal = None
222                    return self.position, self.position
223            except Exception:
224                pass
225
226            # Blocked node?
227            try:
228                if self.city.event_manager.is_node_blocked(nxt):
229                    print(f"[{self.label}]  Node {nxt} blocked  replanning")
230                    self.path = []
231                    self.goal = None
232                    return self.position, self.position
233            except Exception:
234                pass
235
236            old = self.position
237            self.position = nxt
238            self.path = self.path[1:]
239            self.steps_since_plan += 1
240            return old, self.position
241
242        # No movement possible
243        self.path = [self.position]
244        return self.position, self.position

260        """
261        Get the JID of the nearest traffic light.

263        Returns:
264            str: JID of closest traffic-light agent or fallback.

266        lights = list(self.city.traffic_lights.values())

331        async def _wait_for_green(self, from_pos, to_pos):
332            """
333            Request passage permission from the traffic light.
334
335            Args:
336                from_pos (tuple[int, int]): Current position.
337                to_pos (tuple[int, int]): Intended next node.
338
339            Returns:
340                None  waits until 'granted' or keeps retrying.
341            """
342            light_jid = self.agent._light_jid_for(to_pos)
343            if not light_jid:
344                return
345
346            while True:
347                req = Message(to=light_jid)
348                req.set_metadata("type", "passage_request")
349                req.set_metadata("protocol", "contract-net")
350                req.set_metadata("performative", "cfp")
351                req.body = json.dumps({"from": list(from_pos), "to": list(to_pos)})
352                await self.send(req)
353                incoming = await self.receive(timeout=0.8)
354
355                if not incoming:
356                    print(f"[{self.agent.label}]  No reply from {light_jid}, retrying")
357                    await asyncio.sleep(0.4)
358                    continue
359
360                granted_flag = incoming.metadata.get("granted", "false")
361                granted = granted_flag == "true"
362                perf = incoming.metadata.get("performative", "?")
363
364                print(f"[{self.agent.label}]  Reply {incoming.body}, granted={granted_flag}, perf={perf}")
365
366                if granted:
367                    return
368
369                await asyncio.sleep(0.5)
370
371        async def run(self):
372            """
373            Main loop implementing the agent life cycle.
374
375            Handles:
376                - goal arrival,
377                - replanning logic,
378                - incident messages,
379                - traffic light coordination,
380                - collision avoidance,
381                - occupancy + viewer updates.
382            """
383            if self.first_run:
384                await asyncio.sleep(1.5)
385                self.first_run = False
386
387            metrics = getattr(self.agent.city, "metrics", None)
388
389            # -------- Arrival detection --------
390            at_goal = (
391                self.agent.goal is not None
392                and self.agent.position == self.agent.goal
393            )
394            singleton_here = (
395                len(self.agent.path) == 1
396                and self.agent.path[0] == self.agent.position
397            )
398
399            if at_goal or singleton_here:
400                print(f"[{self.agent.label}]  Reached {self.agent.goal}, pausing")
401                if metrics and self.agent.goal is not None:
402                    metrics.end_trip(self.agent.label)
403                await asyncio.sleep(3)
404                self.agent.goal = None
405                self.agent.path = []
406                self.agent.steps_since_plan = 0
407                return
408
409            # -------- Replan conditions --------
410            need_plan = (
411                not self.agent.path
412                or self.agent.goal is None
413                or self.agent.steps_since_plan >= self.agent.replan_every
414            )
415
416            # -------- Incident-based replanning --------
417            incident_near = False
418            try:
419                incident_msg = None
420                if USE_FILTERED_INCIDENTS and Template is not None:
421                    incident_msg = await self.receive(timeout=0.01, filter=Template(metadata={"type": "incident"}))
422                else:
423                    candidate = await self.receive(timeout=0.01)
424                    if candidate and isinstance(candidate.body, str) and "Accident reported near" in candidate.body:
425                        incident_msg = candidate
426
427                if incident_msg and isinstance(incident_msg.body, str):
428                    m = re.search(r"\((\d+),\s*(\d+)\)", incident_msg.body)
429                    if m:
430                        p = (int(m.group(1)), int(m.group(2)))
431                        if manhattan(p, self.agent.position) <= 4:
432                            incident_near = True
433                            print(f"[{self.agent.label}]  Rerouting due to incident at {p}")
434            except Exception as e:
435                print("[INCIDENT CHECK WARN]", e)
436
437            if need_plan or (
438                incident_near
439                and self.agent.goal is not None
440                and self.agent.position != self.agent.goal
441            ):
442                if self.agent.goal is None:
443                    self.agent.goal = self.agent._choose_far_goal()
444                    if metrics:
445                        metrics.start_trip(self.agent.label)
446                self.agent._plan_to(self.agent.goal)
447
448            # -------- Movement + traffic lights --------
449            if self.agent.path:
450                old_pos, candidate_new = self.agent._peek_next_step()
451                if old_pos != candidate_new:
452                    await self._wait_for_green(old_pos, candidate_new)
453
454                    # Collision avoidance
455                    all_vehicles = self.agent.shared.get("vehicles", {})
456                    all_emergency = self.agent.shared.get("emergency", {})
457                    others = {**all_vehicles, **all_emergency}
458                    occupied = any(
459                        pos == candidate_new and name != self.agent.label
460                        for name, pos in others.items()
461                    )
462
463                    if occupied:
464                        print(f"[{self.agent.label}]  Node {candidate_new} occupied  cancelling move")
465                        new_pos = old_pos
466                        self.agent.path = []
467                        self.agent.goal = None
468                        self.agent.steps_since_plan = 0
469                    else:
470                        self.agent.position = candidate_new
471                        if self.agent.position in self.agent.path:
472                            idx = self.agent.path.index(self.agent.position)
473                            self.agent.path = self.agent.path[idx:]
474                        else:
475                            self.agent.path = [self.agent.position]
476
477                        self.agent.steps_since_plan += 1
478                        new_pos = self.agent.position
479                else:
480                    old_pos, new_pos = self.agent._move_randomly()
481            else:
482                old_pos, new_pos = self.agent._move_randomly()
483
484            # -------- Occupancy updates --------
485            try:
486                if old_pos != new_pos:
487                    self.agent.city.occupancy.leave(old_pos, new_pos, self.agent.label)
488                    self.agent.city.occupancy.enter(old_pos, new_pos, self.agent.label)
489
490                if hasattr(self.agent.city, "update_edge_weights"):
491                    self.agent.city.update_edge_weights()
492            except Exception as e:
493                print("[OCCUPANCY/WEIGHTS ERROR]", e)
494
495            # -------- Viewer updates --------
496            try:
497                self.agent.shared.setdefault("vehicles", {})[self.agent.label] = self.agent.position
498            except Exception as e:
499                print("[VIS-WRITE vehicle tick ERROR]", e)
500
501            print(f"[{self.agent.label}]  moved {old_pos}  {new_pos} (goal={self.agent.goal})")

266        """

273                key=lambda l: nx.shortest_path_length(self.city.graph, self.position, l)

288                idx = self.path.index(self.position)
289                self.path = self.path[idx:]
290            else:
291                return self.position, self.position
292
293        if len(self.path) >= 2:
294            return self.position, self.path[1]
295        return self.position, self.position

335
336            Args:
337                from_pos (tuple[int, int]): Current position.
338                to_pos (tuple[int, int]): Intended next node.
339
340            Returns:
341                None  waits until 'granted' or keeps retrying.
342            """

359                    continue

361                granted_flag = incoming.metadata.get("granted", "false")
362                granted = granted_flag == "true"
363                perf = incoming.metadata.get("performative", "?")
364
365                print(f"[{self.agent.label}]  Reply {incoming.body}, granted={granted_flag}, perf={perf}")
366
367                if granted:
368                    return

372        async def run(self):
373            """
374            Main loop implementing the agent life cycle.
375
376            Handles:
377                - goal arrival,
378                - replanning logic,
379                - incident messages,
380                - traffic light coordination,
381                - collision avoidance,
382                - occupancy + viewer updates.
383            """
384            if self.first_run:
385                await asyncio.sleep(1.5)
386                self.first_run = False
387
388            metrics = getattr(self.agent.city, "metrics", None)
389
390            # -------- Arrival detection --------
391            at_goal = (
392                self.agent.goal is not None
393                and self.agent.position == self.agent.goal
394            )
395            singleton_here = (
396                len(self.agent.path) == 1
397                and self.agent.path[0] == self.agent.position
398            )
399
400            if at_goal or singleton_here:
401                print(f"[{self.agent.label}]  Reached {self.agent.goal}, pausing")
402                if metrics and self.agent.goal is not None:
403                    metrics.end_trip(self.agent.label)
404                await asyncio.sleep(3)
405                self.agent.goal = None
406                self.agent.path = []
407                self.agent.steps_since_plan = 0
408                return
409
410            # -------- Replan conditions --------
411            need_plan = (
412                not self.agent.path
413                or self.agent.goal is None
414                or self.agent.steps_since_plan >= self.agent.replan_every
415            )
416
417            # -------- Incident-based replanning --------
418            incident_near = False
419            try:
420                incident_msg = None
421                if USE_FILTERED_INCIDENTS and Template is not None:
422                    incident_msg = await self.receive(timeout=0.01, filter=Template(metadata={"type": "incident"}))
423                else:
424                    candidate = await self.receive(timeout=0.01)
425                    if candidate and isinstance(candidate.body, str) and "Accident reported near" in candidate.body:
426                        incident_msg = candidate
427
428                if incident_msg and isinstance(incident_msg.body, str):
429                    m = re.search(r"\((\d+),\s*(\d+)\)", incident_msg.body)
430                    if m:
431                        p = (int(m.group(1)), int(m.group(2)))
432                        if manhattan(p, self.agent.position) <= 4:
433                            incident_near = True
434                            print(f"[{self.agent.label}]  Rerouting due to incident at {p}")
435            except Exception as e:
436                print("[INCIDENT CHECK WARN]", e)
437
438            if need_plan or (
439                incident_near
440                and self.agent.goal is not None
441                and self.agent.position != self.agent.goal
442            ):
443                if self.agent.goal is None:
444                    self.agent.goal = self.agent._choose_far_goal()
445                    if metrics:
446                        metrics.start_trip(self.agent.label)
447                self.agent._plan_to(self.agent.goal)
448
449            # -------- Movement + traffic lights --------
450            if self.agent.path:
451                old_pos, candidate_new = self.agent._peek_next_step()
452                if old_pos != candidate_new:
453                    await self._wait_for_green(old_pos, candidate_new)
454
455                    # Collision avoidance
456                    all_vehicles = self.agent.shared.get("vehicles", {})
457                    all_emergency = self.agent.shared.get("emergency", {})
458                    others = {**all_vehicles, **all_emergency}
459                    occupied = any(
460                        pos == candidate_new and name != self.agent.label
461                        for name, pos in others.items()
462                    )
463
464                    if occupied:
465                        print(f"[{self.agent.label}]  Node {candidate_new} occupied  cancelling move")
466                        new_pos = old_pos
467                        self.agent.path = []
468                        self.agent.goal = None
469                        self.agent.steps_since_plan = 0
470                    else:
471                        self.agent.position = candidate_new
472                        if self.agent.position in self.agent.path:
473                            idx = self.agent.path.index(self.agent.position)
474                            self.agent.path = self.agent.path[idx:]
475                        else:
476                            self.agent.path = [self.agent.position]
477
478                        self.agent.steps_since_plan += 1
479                        new_pos = self.agent.position
480                else:
481                    old_pos, new_pos = self.agent._move_randomly()
482            else:
483                old_pos, new_pos = self.agent._move_randomly()
484
485            # -------- Occupancy updates --------
486            try:
487                if old_pos != new_pos:
488                    self.agent.city.occupancy.leave(old_pos, new_pos, self.agent.label)
489                    self.agent.city.occupancy.enter(old_pos, new_pos, self.agent.label)
490
491                if hasattr(self.agent.city, "update_edge_weights"):
492                    self.agent.city.update_edge_weights()
493            except Exception as e:
494                print("[OCCUPANCY/WEIGHTS ERROR]", e)
495
496            # -------- Viewer updates --------
497            try:
498                self.agent.shared.setdefault("vehicles", {})[self.agent.label] = self.agent.position
499            except Exception as e:
500                print("[VIS-WRITE vehicle tick ERROR]", e)
501
502            print(f"[{self.agent.label}]  moved {old_pos}  {new_pos} (goal={self.agent.goal})")
