agents.emergency_vehicle

Emergency vehicle agent module.

This module defines the EmergencyVehicleAgent, a SPADE autonomous agent that implements intelligent emergency response behavior in the simulation.

Key features: - A* routing with reduced congestion penalties (faster navigation) - Avoidance of blocked edges reported by EventManager - Automatic switching between "to_hospital" and "to_random" phases - Traffic-light priority negotiation - Integration with the Metrics system to track emergency response time - Incident gossip handling (merging distributed incident information)

This agent continuously cycles: hospital → random incident → hospital → random incident → ...

119    def _dynamic_weight(self, u, v, d):
120        """
121        Compute the dynamic weight of a road segment for A* routing.
122
123        Emergency vehicles have lighter congestion penalties than normal cars
124        but still respect incident penalties.
125
126        Args:
127            u (tuple): Starting node.
128            v (tuple): Ending node.
129            d (dict): Edge attribute dictionary.
130
131        Returns:
132            float: Effective weight.
133        """
134        base = d.get("weight", 1.0)
135
136        try:
137            occ = self.city.occupancy.edge_density(u, v)
138        except Exception:
139            occ = 0.0
140
141        try:
142            pen = self.city.event_manager.edge_penalty(u, v)
143        except Exception:
144            pen = 0.0
145
146        return base + 0.3 * occ + pen

153            tuple | None: (x, y) of the closest hospital, or None.
154        """
155        hospitals = getattr(self.city, "hospitals", {}).values()
156        if not hospitals:
157            return None

168            min_manhattan (int): Required Manhattan distance.

171            tuple: Selected node coordinate.

182    def _plan_to(self, dest):
183        """
184        Plan a path to the given destination using A*.

186        Args:
187            dest (tuple): Target node coordinate.

190        if metrics:

193        try:

195                self.city.graph,
196                self.position,

206            self.path = []

209    def _step_along_path(self):
210        """
211        Move one step along the planned A* path.
212
213        Returns:
214            tuple: (old_pos, new_pos)
215        """
216        if not self.path or self.path[0] != self.position:
217            if self.path and self.position in self.path:
218                i = self.path.index(self.position)
219                self.path = self.path[i:]
220            else:
221                return self._move_randomly()
222
223        if len(self.path) >= 2:
224            curr = self.path[0]
225            nxt = self.path[1]
226
227            # Blocked edge check
228            try:
229                if self.city.event_manager.is_blocked((curr, nxt)):
230                    print(f"[{self.label}] ⛔ Edge {(curr, nxt)} blocked, waiting and replanning")
231                    self.path = []
232                    self.goal = None
233                    return self.position, self.position
234            except Exception:
235                pass
236
237            old = self.position
238            self.position = nxt
239            self.path = self.path[1:]
240            self.steps_since_plan += 1
241            return old, self.position
242
243        # At destination or path exhausted
244        return self.position, self.position

246    def _move_randomly(self):
247        """
248        Take a random step when no path exists.
249
250        Returns:
251            tuple: (old_pos, new_pos)
252        """
253        nbrs = list(self.city.graph.neighbors(self.position))
254        old = self.position
255        if nbrs:
256            self.position = random.choice(nbrs)
257        return old, self.position

260        """
261        Determine the JID of the nearest traffic light.

263        Returns:
264            str: Traffic light JID.
265        """
266        lights = list(self.city.traffic_lights.values())
267        if not lights:

270            best = min(
271                lights,
272                key=lambda l: nx.shortest_path_length(self.city.graph, self.position, l),

281    class EmergencyBehaviour(behaviour.PeriodicBehaviour):
282        """
283        Periodic behaviour controlling movement, replanning, incident gossip
284        ingestion, priority negotiation, emergency timing, and collision
285        avoidance for the emergency vehicle.
286        """
287
288        async def on_start(self):
289            """Executed once when the behaviour begins."""
290            self.first_run = True

292        async def run(self):
293            """
294            Main loop executed periodically:
295                - read incident gossip
296                - perform arrival checks and phase switching
297                - plan new routes when needed
298                - step along A* path
299                - manage collisions
300                - request priority from nearest traffic light
301                - log emergency start/end for metrics
302            """
303            if self.first_run:
304                await asyncio.sleep(1.0)
305                self.first_run = False
306
307            metrics = getattr(self.agent.city, "metrics", None)
308            hospitals = getattr(self.agent.city, "hospitals", {})
309            hospital_positions = set(hospitals.values()) if hospitals else set()
310
311            # --------------------------------------------------------------
312            # Merge incoming incident gossip (JSON messages)
313            # --------------------------------------------------------------
314            incoming = await self.receive(timeout=0.2)
315
316            if incoming and incoming.body:
317                try:
318                    data = json.loads(incoming.body)
319                except Exception:
320                    data = None
321
322                if isinstance(data, dict) and data.get("type") == "incident_gossip":
323                    payload = data.get("payload")
324                    if payload:
325                        try:
326                            self.agent.city.event_manager.merge_from_message(payload)
327                            print(f"[{self.agent.label}] 📩 Merged incident update: {payload}")
328                        except Exception as e:
329                            print("[EMERGENCY] Failed to merge incident:", e)
330
331            # --------------------------------------------------------------
332            # Arrival check (hospital or random target)
333            # --------------------------------------------------------------
334            at_goal = self.agent.goal and self.agent.position == self.agent.goal
335            singleton_here = (
336                len(self.agent.path) == 1
337                and self.agent.path[0] == self.agent.position
338            )
339
340            if at_goal or singleton_here:
341                # Record EV response completion
342                if metrics:
343                    if self.agent.fixed_dest:
344                        if (
345                            self.agent.position == self.agent.fixed_dest
346                            and self.agent.phase == "to_hospital"
347                        ):
348                            metrics.end_emergency()
349                    else:
350                        if (
351                            hospital_positions
352                            and self.agent.position in hospital_positions
353                            and self.agent.phase == "to_hospital"
354                        ):
355                            metrics.end_emergency()
356
357                await asyncio.sleep(self.agent.pause_at_goal)
358
359                # Phase switch
360                if self.agent.fixed_dest:
361                    self.agent.phase = (
362                        "to_random"
363                        if self.agent.position == self.agent.fixed_dest
364                        else "to_hospital"
365                    )
366                else:
367                    self.agent.phase = (
368                        "to_random"
369                        if hospital_positions and self.agent.position in hospital_positions
370                        else "to_hospital"
371                    )
372
373                self.agent.goal = None
374                self.agent.path = []
375                self.agent.steps_since_plan = 0
376                return
377
378            # --------------------------------------------------------------
379            # Route planning ⇢ choose destination and maybe start emergency
380            # --------------------------------------------------------------
381            if not self.agent.path or self.agent.goal is None:
382                dest = None
383
384                if self.agent.fixed_dest:
385                    # Hospital <-> random
386                    dest = (
387                        self.agent.fixed_dest
388                        if self.agent.phase == "to_hospital"
389                        else self.agent._choose_far_goal()
390                    )
391
392                    if (
393                        metrics
394                        and self.agent.position == self.agent.fixed_dest
395                        and self.agent.phase == "to_random"
396                    ):
397                        metrics.start_emergency()
398
399                else:
400                    # Any hospital works
401                    dest = (
402                        self.agent._closest_hospital()
403                        if self.agent.phase == "to_hospital"
404                        else self.agent._choose_far_goal()
405                    )
406
407                    if (
408                        metrics
409                        and hospital_positions
410                        and self.agent.position in hospital_positions
411                        and self.agent.phase == "to_random"
412                    ):
413                        metrics.start_emergency()
414
415                if dest:
416                    self.agent._plan_to(dest)
417
418            # --------------------------------------------------------------
419            # Move one planned step
420            # --------------------------------------------------------------
421            old_pos, new_pos = self.agent._step_along_path()
422
423            # --------------------------------------------------------------
424            # Collision avoidance:
425            #   - avoid cars
426            #   - avoid other ambulances unless arriving at hospital
427            # --------------------------------------------------------------
428            all_vehicles = self.agent.shared.get("vehicles", {})
429            all_emergency = self.agent.shared.get("emergency", {})
430
431            at_hospital_node = (
432                (self.agent.fixed_dest and new_pos == self.agent.fixed_dest)
433                or (
434                    not self.agent.fixed_dest
435                    and hospital_positions
436                    and new_pos in hospital_positions
437                )
438            )
439
440            occupied_by_vehicle = any(pos == new_pos for pos in all_vehicles.values())
441            occupied_by_other_ambulance = any(
442                pos == new_pos and name != self.agent.label
443                for name, pos in all_emergency.items()
444            )
445
446            must_block = (
447                occupied_by_vehicle
448                or (occupied_by_other_ambulance and not at_hospital_node)
449            )
450
451            if must_block:
452                print(f"[{self.agent.label}] ⛔ Target {new_pos} occupied, staying at {old_pos}")
453                self.agent.position = old_pos
454                new_pos = old_pos
455                self.agent.path = []
456                self.agent.goal = None
457                self.agent.steps_since_plan = 0
458
459            # --------------------------------------------------------------
460            # Occupancy + visualization update
461            # --------------------------------------------------------------
462            try:
463                if old_pos != new_pos:
464                    self.agent.city.occupancy.leave(old_pos, new_pos, self.agent.label)
465                    self.agent.city.occupancy.enter(old_pos, new_pos, self.agent.label)
466                self.agent.shared.setdefault("emergency", {})[self.agent.label] = self.agent.position
467            except Exception as e:
468                print("[EMERGENCY UPDATE WARN]", e)
469
470            # --------------------------------------------------------------
471            # Priority request to traffic light
472            # --------------------------------------------------------------
473            light_jid = self.agent._nearest_light_jid()
474
475            req = Message(to=light_jid)
476            req.set_metadata("type", "priority_request")
477            req.body = json.dumps({"from": list(old_pos), "to": list(new_pos)})
478
479            await self.send(req)
480
481            incoming = await self.receive(timeout=0.6)
482            if not incoming:
483                await asyncio.sleep(0.3)
484                await self.send(req)
485                incoming = await self.receive(timeout=0.6)
486
487            granted_flag = (
488                incoming.metadata.get("granted", "false")
489                if incoming and incoming.metadata
490                else "false"
491            )
492            print(
493                f"[{self.agent.label}] 📩 Traffic light reply: "
494                f"body={incoming.body if incoming else None}, granted={granted_flag}"
495            )

Features: - Avoids blocked edges reported by EventManager. - Negotiates priority with traffic lights. - Uses congestion-aware edge weights. - Sends/receives incident gossip updates. - Triggers metric collection for emergency response time.

476            req = Message(to=light_jid)
477            req.set_metadata("type", "priority_request")
478            req.body = json.dumps({"from": list(old_pos), "to": list(new_pos)})
479
480            await self.send(req)
481
482            incoming = await self.receive(timeout=0.6)
483            if not incoming:
484                await asyncio.sleep(0.3)
485                await self.send(req)
486                incoming = await self.receive(timeout=0.6)
487
488            granted_flag = (
489                incoming.metadata.get("granted", "false")
490                if incoming and incoming.metadata
491                else "false"
492            )
493            print(
494                f"[{self.agent.label}] 📩 Traffic light reply: "
495                f"body={incoming.body if incoming else None}, granted={granted_flag}"
